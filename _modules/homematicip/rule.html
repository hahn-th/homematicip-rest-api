    def set_label(self, label):
        """ sets the label of the rule """
        data = {"ruleId": self.id, "label": label}
        return self._restCall("rule/setRuleLabel", json.dumps(data))



    def __str__(self):
        return "{} {} active({})".format(self.ruleType, self.label, self.active)
